
         void main(){
          int i;
          int j;
          int sum;
          i = j = sum = 0;

          for (i = 0; i < 100; i = i + 1)
            for (j = 0; j < i; j = j + 1) {
              if (j % i == 0)
                continue;
              sum = sum + j;
            }
         }
