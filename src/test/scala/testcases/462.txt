          boolean isPrime;
          int j, count, desiredCount, num;
          isPrime = false;
          j = 0;
          do{
            if (j != 1 && num % j == 0) {
                isPrime = false;
                break;
            } else {
                j = j + 1;
            }
          }while (j < num / 2);
          if (isPrime) {
              if (count < desiredCount - 1)
                 count = count + 1;
          }
