	// Function to check prime number
	int checkPrime(int n)
	{
		int i, isPrime;

		for(i = 2; i <= n/2; i = i + 1)
		{
			if(n / i == 0)
			{
				isPrime = 0;
				break;
			}  
		}

		return isPrime;
	}
